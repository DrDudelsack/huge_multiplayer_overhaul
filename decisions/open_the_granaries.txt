country_decisions = {
	open_the_granaries = {
		potential = {
			NOT = {
				has_variable = open_granaries
				is_ai = yes
			}
			num_of_cities >= 1
		}
		allow = {
			any_allied_country = {
				any_countries_at_war_with = {
					war_with = ROOT
				}
			}
		}
		effect = {
			trigger_event = {
				id = OTG.10
			}

			set_variable = {
				name = open_granaries
				value = yes
			}
		}

	}
	close_the_granaries = {
		potential = {
			has_variable = open_granaries
		}
		effect = {
			remove_variable = open_granaries
			trigger_event = {
				id = OTG.2
			}
		}
	}

}
