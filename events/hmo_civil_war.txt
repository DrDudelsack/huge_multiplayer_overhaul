namespace = hmo_civil_war

hmo_civil_war.1 = {
	type = country_event
	hidden = yes

	trigger = {
		is_ai = no
		has_civil_war = yes
	}

	immediate = {
		every_country_state = {
			if = {
				limit = {
					owner = ROOT
					state_loyalty < 50
				}
			}

			#state scope
			add_state_modifier = {
				name = civil_war_unrest
				duration = 720
			}

			random_state_province = {
				if = {
					limit = {
						owner = ROOT
						is_state_capital = yes
					}
				}

				#for some reason this is province scope
				add_state_loyalty = -25
			}
		}
	}
}
