namespace = country_rank

# sets migrant horde modifiers
country_rank.1 = {
	type = country_event
	hidden = yes

	trigger = {
		rank = migrant_horde
		NOT = {
			has_country_modifier = migrant_horde_ae
		}
	}

	immediate = {
		if = {
			limit = {
				NOT = {
					has_country_modifier = migrant_horde_ae
				}
			}

			trigger_event = {
				id = country_rank.7
			}

			add_country_modifier = {
				name = migrant_horde_ae
				duration = -1
			}
		}
	}
}

# remove all modifiers common
country_rank.7 = {
	type = country_event
	hidden = yes

	immediate = {
		if = {
			limit = {
				has_country_modifier = migrant_horde_ae
			}

			remove_country_modifier = migrant_horde_ae
