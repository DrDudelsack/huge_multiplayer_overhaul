TTC.2 = {
	type = country_event
	hidden = yes

	trigger = {
		any_owned_province = {
			NOT = {
				is_port = yes
				can_have_port = yes
			}
			OR = {
				has_province_modifier = MarketPort
				has_province_modifier = FishingPort
			}
			
		}
	}
	immediate = {
		every_owned_province = {
			limit = {
				NOT = {
					is_port = yes
					can_have_port = yes
				}
				OR = {
					has_province_modifier = MarketPort
					has_province_modifier = FishingPort
				}
			}
			if = {
				limit = {
					has_province_modifier = MarketPort
				}
				remove_province_modifier = MarketPort
			}
			if = {
				limit = {
					has_province_modifier = FishingPort
				}
				remove_province_modifier = FishingPort
			}
		}
	}
}
