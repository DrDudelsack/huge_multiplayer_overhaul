namespace = invention

invention.1 = {
	type = country_event
	hidden = yes

	trigger = {
		NOT = {
			has_variable = inv_check
		}
		
		OR = {
			mil_tech_check = 0
			ora_tech_check = 0
			civ_tech_check = 0
			rel_tech_check = 0
		}
	}

	immediate = {
		if = {
			limit = {
				NOT = { has_variable = inv_check }
				OR = {
					military_tech = 0
					OR = {
						military_tech >= 5
						oratory_tech >= 5
						civic_tech >= 5
						religious_tech >= 5
					}
				}

			}

			add_innovation = 4

			set_variable = {
				name = inv_check
				value = 1
			}
		}
		
		if = {
			limit = {
				mil_tech_check > 0
				ora_tech_check > 0
				civ_tech_check > 0
				rel_tech_check > 0
			}

			remove_variable = inv_check
		}
	}
}
