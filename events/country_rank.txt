namespace = country_rank

# sets migrant horde modifiers
country_rank.1 = {
	type = country_event
	hidden = yes

	trigger = {
		rank = migrant_horde
		NOT = {
			has_country_modifier = migrant_horde_ae
		}
	}

	immediate = {
		trigger_event = {
			id = country_rank.7
		}

		add_country_modifier = migrant_horde_ae
	}
}

# remove all modifiers common
country_rank.7 = {
	type = country_event
	hidden = yes

	immediate = {
		if = {
			limit = {
				has_country_modifier = migrant_horde_ae
			}

			remove_country_modifier = migrant_horde_ae
		}

		if = {
			limit = {
				has_country_modifier = city_power_ae
			}

			remove_country_modifier = city_power_ae
		}

		if = {
			limit = {
				has_country_modifier = local_power_ae
			}

			remove_country_modifier = local_power_ae
		}

		if = {
			limit = {
				has_country_modifier = regional_power_ae
			}

			remove_country_modifier = regional_power_ae
		}

		if = {
			limit = {
				has_country_modifier = major_power_ae
			}

			remove_country_modifier = major_power_ae
		}

		if = {
			limit = {
				has_country_modifier = great_power_ae
			}

			remove_country_modifier = great_power_ae
		}
	}
}
