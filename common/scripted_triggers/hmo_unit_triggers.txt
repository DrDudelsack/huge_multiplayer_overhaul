can_assault_port = {
	is_sea = no
	has_owner = yes
	is_port = yes
	fort_level >= 1
	NOT = { controller = scope:naval_assaulter_unit.unit_owner }
	controller = { war_with = scope:naval_assaulter_unit.unit_owner }
	NOT = { has_enemy_army = scope:naval_assaulter_unit.unit_owner }
}

can_raid_port = {
	is_sea = no
	has_owner = yes
	is_port = yes
	NOT = { controller = scope:naval_raider_unit.unit_owner }
	controller = { war_with = scope:naval_raider_unit.unit_owner }
	has_siege = yes
	siege = {
		has_breach = no
	}
}

trading_partner_province_trigger = {
	has_owner = yes
	is_port = yes
	NOT = {
		OR = {
			has_province_modifier = mugged_by_reality 
			has_province_modifier = trading_partner
		} 
	}
	owner = {
		OR = {
			exports_to = root.unit_owner
			imports_from = root.unit_owner
		}
	}
	owner = controller
		NOT = {
			owner = root.unit_owner
		}
}

fishing_province_trigger = {
	has_owner = yes
	is_sea = no
	NOT = {
		has_province_modifier = getting_fished
	}
	total_population > 4
	trade_goods = fish	
}

can_mug_province_trigger = {
	is_sea = no
	has_owner = yes
	is_port = yes
	NOT = { owner = root.unit_owner }
	trigger_if = {
		limit = { root.unit_owner = { is_subject = yes } }
		NOT = { root.unit_owner.overlord = this.owner }
	}
	owner = {
		NOT = { alliance_with = root.unit_owner }
		NOT = { is_subject_of = root.unit_owner }
		NOT = { has_truce_with = root.unit_owner }
		NOT = { exports_to = root.unit_owner }
		NOT = { imports_from = root.unit_owner }
	}
	NOT = {
		OR = {
			has_province_modifier = mugged_by_reality 
			has_province_modifier = trading_partner
		} 
	}
	owner = controller
}

naval_supply_trigger = {
	has_owner = yes
	is_port = yes
	NOT = {
		has_province_modifier = supplied_by_sea
	}
	OR = {
		owner - unit_owner
		controller = unit_owner
	}
}
