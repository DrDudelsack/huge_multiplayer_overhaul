		}

			is_alive = yes
			is_mercenary = no
			employer = {

					tag = PIR

			}

	}

					NOT = { has_variable = anti_spam_hunt }
				}
			}
			NOT = {
				has_variable = hunting_friend
			}		
		}
		scope:actor = {
			in_diplomatic_range = scope:target.employer
			can_pay_price = gift_slave
		}
		scope:actor.current_ruler = {
			NOT = {

			}

	}

			}
