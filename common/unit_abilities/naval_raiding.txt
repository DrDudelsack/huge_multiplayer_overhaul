naval_raiding = {
	duration = 31
	enable = yes
	toggle = no
	navy_only = yes
	cancel_on_combat_end = yes

	map = yes
	
	hidden = {
		unit_owner = {
			OR = {
				num_of_cities < 1
				NOR = {
					heritage = pirate_heritage
					is_tradition_tree_allowed = naval_traditions
				}
			}
		}
	}

	allow = {
		unit_owner = {
			trigger_if = {
				limit = {
					heritage = pirate_heritage
				}
				heritage = pirate_heritage
			}
			trigger_else_if = {
				limit = {
					is_tradition_tree_allowed = naval_traditions
				}
				OR = {
					has_military_bonus = greek_antigonid_path_4
					has_military_bonus = aryan_river_path_3
					has_military_bonus = illyrian_piracy_path_1
					has_military_bonus = lydian_path_1
					has_military_bonus = celt_iberian_path_c_3
					has_military_bonus = arabian_path_4
				}
				
			}
			trigger_else = {
				always = no
			}
		}
		unit_location = {
			custom_tooltip = {
				text = naval_raiding_sea_province
				OR = {
					terrain = riverine_terrain
					terrain = ocean
					terrain = coastal_terrain
					terrain = tributary_terrain
				}
			}
		}
		is_navy = yes
		unit_size >= 10
		in_combat = no
		is_moving = no
		in_siege = no
		is_exiled = no
		NOT = { unit_owner = { has_low_economic_policy = expense_navy } }
		trigger_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							naval_raid_target_province_trigger = yes
						}
					}
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						total_population > 5
						fort_level <= 0
						NOT = { owner = root.unit_owner }
						root.unit_owner = { is_subject = yes }
						root.unit_owner.overlord = this.owner
						owner = {
							NOT = { alliance_with = root.unit_owner }
							NOT = { is_subject_of = root.unit_owner }
							NOT = { has_truce_with = root.unit_owner }
						}
						is_port = yes
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
			custom_tooltip = {
				text = naval_raiding_only_overlord_ports_tooltip
				always = no
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							naval_raid_target_province_trigger = yes
						}
					}
				}
			}
			custom_tooltip = {
				text = explain_naval_raiding_tooltip
				always = no
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							naval_raid_target_province_trigger = yes
							NOT = {
								has_province_modifier = port_province_raided
							}
							NOT = { has_enemy_army = root.unit_owner }
						}
					}
				}
			}
			custom_tooltip = {
				text = naval_raiding_cooldown_tooltip
				always = no
			}
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
	}
	
	start_effect = {
		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						naval_raid_target_province_trigger = yes
					}
					NOT = { has_enemy_army = root.unit_owner }
				}
			}

			hidden_effect = {
				save_scope_as = slave_raiding_unit
				unit_location = {
					save_scope_as = slave_raiding_location
				}
				set_variable = is_slave_raiding

				unit_location = {
					trigger_event = {
						id = hmo_unit_abilities.7
					}
				}
			}

			add_unit_modifier = {
				name = naval_action
				duration = 30
			}

			custom_tooltip = naval_raiding_ae_gain_tooltip
			show_as_tooltip = { #All fake for the tooltips sake
				unit_location = {
					every_neighbor_province = {
						limit = {
							naval_raid_target_province_trigger = yes
							NOT = {	has_province_modifier = port_province_raided }
							NOT = { has_enemy_army = root.unit_owner }
						}
						custom = naval_raiding_explain_modifier_tooltip
						add_province_modifier = {
							name = port_province_raided
							duration = 3650
						}
					}
				}
			}
			hidden_effect = {
				root.unit_owner = {
					add_aggressive_expansion = 1
				}
				trigger_event = {
					id = hmo_unit_abilties.1
					days = 30
				}
			}
		}
		else = {
			custom_tooltip = naval_raiding_tooltip
		}
	}

	finish_effect = {
		remove_variable = is_slave_raiding
	}

	finished_when = {
		OR = {
			in_combat = yes
			is_moving = yes
			is_exiled = yes
			unit_owner = { has_low_economic_policy = expense_navy }
			has_commander = no
			navy_drilling = yes
			navy_supplying = yes
			navy_assaulting = yes
			navy_fishing = yes
			navy_raiding = yes
			navy_raiding_commerce = yes
			navy_protecting_commerce = yes
		}
	}

	idle_entity_state = combat
}
