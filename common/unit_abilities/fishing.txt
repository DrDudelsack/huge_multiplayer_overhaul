fishing = {
	duration = -1
	enable = yes
	toggle = yes
	navy_only = yes
	cancel_on_combat_end = yes

	map = yes

	soundeffect ="event:/SFX/UI/Unit/sfx_ui_unit_reorganization"

	allow = {
		is_navy = yes
		unit_size >= 10
		in_combat = no
		is_moving = no
		is_exiled = no
		in_siege = no
		has_commander = yes
		
		trigger_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							fishing_province_trigger = yes
						}
					}
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						trade_goods = fish
						NOT = {
							has_province_modifier = getting_fished
						}
					}
				}
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							fishing_province_trigger = yes
						}
					}
				}
			}
			custom_tooltip = {
				text = explain_fishing_tooltip
				always = no
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							fishing_province_trigger = yes
							NOT = {
								has_province_modifier = getting_fished
							}
						}
					}
				}
			}
			custom_tooltip = {
				text = fishing_cooldown_tooltip
				always = no
			}
		}

		NOT = { unit_owner = { has_country_modifier = go_fish } }
		NOT = { unit_owner = { has_low_economic_policy = expense_navy } }		
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
		unit_location = {
			any_neighbor_province = {
				is_sea = no
				fishing_province_trigger = yes
				NOT = {
					has_province_modifier = getting_fished
				}
			}
		}
	}


	start_effect = {
		add_unit_modifier = {
			name = naval_action
			duration = -1
		}

		hidden_effect = {
			set_variable = is_navy_fishing			
		}

		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						fishing_province_trigger = yes
					}
				}
			}
			show_as_tooltip = { 
				unit_location = {
					every_neighbor_province = { 
						limit = {
							fishing_province_trigger = yes
							NOT = {	has_province_modifier = getting_fished }
						}
						custom = fishing_explain_modifier_tooltip
						add_province_modifier = {
							name = getting_fished
							duration = -1
						}
					}
				}
			}

			unit_location = {
				ordered_neighbor_province = {
					limit = {
						fishing_province_trigger = yes
						NOT = {	has_province_modifier = getting_fished }
					}
					order_by = total_population
					hidden_effect = {
						add_province_modifier = {
							name = getting_fished
							duration = -1
						}
					}
				}
			}
		}

	}

	finished_when = {
		OR = {
			is_exiled = yes
			unit_owner = { has_low_economic_policy = expense_navy }
			is_moving = yes
			has_commander = no
			strength_percentage < 0.3
			morale_percentage < 0.3
			navy_drilling = yes
			navy_supplying = yes
		}
	}

	finish_effect = {
		unit_location = {
			any_neighbor_province = {
				limit = {
					has_province_modifier = getting_fished
				}
				remove_province_modifier = getting_fished
			}
		}
		unit_owner = {
			remove_country_modifier = go_fish
		}
		remove_variable = is_navy_fishing
	}
	
	ai_will_do = {
		modifier = {
			trigger = {

				experience_percentage < 0.65
				
				unit_owner = {
					treasury > 50
					manpower > 20
				}
			}
			add = {
				value = 250
				divide = unit_size_rank
			}
		}
	}
	
	ai_will_revoke = {
		OR = {
			is_idle = no
			unit_threat > 0.2
			experience_percentage >= 0.95

			unit_owner = {
				treasury < 1
			}
			unit_owner = {
				months_to_war < 12
			}
		}
	}
}
